50 void
51 AUXADCDisable(void)
52 {
53  // Disable the ADC reference
54  ADI8BitsClear(AUX_ADI4_BASE, ADI_4_AUX_O_ADCREF0, ADI_4_AUX_ADCREF0_EN_M | ADI_4_AUX_ADCREF0_REF_ON_IDLE_M | ADI_4_AUX_ADCREF0_SRC_M);
55 
56  // Assert reset and disable the ADC
57  ADI8BitsClear(AUX_ADI4_BASE, ADI_4_AUX_O_ADC0, ADI_4_AUX_ADC0_EN_M | ADI_4_AUX_ADC0_RESET_N_M | ADI_4_AUX_ADC0_SMPL_MODE_M | ADI_4_AUX_ADC0_SMPL_CYCLE_EXP_M);
58 
59  // Disable the ADC clock (no need to wait since IOB_WUC_ADCCLKCTL_ACK goes low immediately)
60  HWREG(AUX_WUC_BASE + AUX_WUC_O_ADCCLKCTL) = 0;
61 
62  // Disable the ADC data interface
63  HWREG(AUX_ANAIF_BASE + AUX_ANAIF_O_ADCCTL) = 0;
64 }

66 //*****************************************************************************
67 //
68 // Enables the ADC for asynchronous operation
69 //
70 //*****************************************************************************
71 void
72 AUXADCEnableAsync(uint32_t refSource, uint32_t trigger)
73 {
74  // Enable the ADC reference, with the following options:
75  // - SRC: Set when using relative reference
76  // - REF_ON_IDLE: Always cleared since there is no idle state in asynchronous operation
77  ADI8BitsSet(AUX_ADI4_BASE, ADI_4_AUX_O_ADCREF0, refSource | ADI_4_AUX_ADCREF0_EN_M);
78 
79  // Enable the ADC clock
80  HWREG(AUX_WUC_BASE + AUX_WUC_O_ADCCLKCTL) = AUX_WUC_ADCCLKCTL_REQ_M;
81  while (!(HWREG(AUX_WUC_BASE + AUX_WUC_O_ADCCLKCTL) & AUX_WUC_ADCCLKCTL_ACK_M));
82 
83  // Enable the ADC data interface
84  if (trigger == AUXADC_TRIGGER_MANUAL) {
85  // Manual trigger: No need to configure event routing from GPT
86  HWREG(AUX_ANAIF_BASE + AUX_ANAIF_O_ADCCTL) = AUX_ANAIF_ADCCTL_START_SRC_NO_EVENT0 | AUX_ANAIF_ADCCTL_CMD_EN;
87  } else {
88  // GPT trigger: Configure event routing via MCU_EV to the AUX domain
89  HWREG(EVENT_BASE + EVENT_O_AUXSEL0) = trigger;
90  HWREG(AUX_ANAIF_BASE + AUX_ANAIF_O_ADCCTL) = AUX_ANAIF_ADCCTL_START_SRC_MCU_EV | AUX_ANAIF_ADCCTL_CMD_EN;
91  }
92 
93  // Release reset and enable the ADC
94  ADI8BitsSet(AUX_ADI4_BASE, ADI_4_AUX_O_ADC0, ADI_4_AUX_ADC0_EN_M | ADI_4_AUX_ADC0_RESET_N_M | ADI_4_AUX_ADC0_SMPL_MODE_M);
95 }

97 //*****************************************************************************
98 //
99 // Enables the ADC for synchronous operation
100 //
101 //*****************************************************************************
102 void
103 AUXADCEnableSync(uint32_t refSource, uint32_t sampleTime, uint32_t trigger)
104 {
105  // Enable the ADC reference, with the following options:
106  // - SRC: Set when using relative reference
107  // - REF_ON_IDLE: Set when using fixed reference and sample time < 6
108  uint8_t adcref0 = refSource | ADI_4_AUX_ADCREF0_EN_M;
109  if (!refSource && (sampleTime < 6)) {
110  adcref0 |= ADI_4_AUX_ADCREF0_REF_ON_IDLE_M;
111  }
112  ADI8BitsSet(AUX_ADI4_BASE, ADI_4_AUX_O_ADCREF0, adcref0);
113 
114  // Enable the ADC clock
115  HWREG(AUX_WUC_BASE + AUX_WUC_O_ADCCLKCTL) = AUX_WUC_ADCCLKCTL_REQ_M;
116  while (!(HWREG(AUX_WUC_BASE + AUX_WUC_O_ADCCLKCTL) & AUX_WUC_ADCCLKCTL_ACK_M));
117 
118  // Enable the ADC data interface
119  if (trigger == AUXADC_TRIGGER_MANUAL) {
120  // Manual trigger: No need to configure event routing from GPT
121  HWREG(AUX_ANAIF_BASE + AUX_ANAIF_O_ADCCTL) = AUX_ANAIF_ADCCTL_START_SRC_NO_EVENT0 | AUX_ANAIF_ADCCTL_CMD_EN;
122  } else {
123  // GPT trigger: Configure event routing via MCU_EV to the AUX domain
124  HWREG(EVENT_BASE + EVENT_O_AUXSEL0) = trigger;
125  HWREG(AUX_ANAIF_BASE + AUX_ANAIF_O_ADCCTL) = AUX_ANAIF_ADCCTL_START_SRC_MCU_EV | AUX_ANAIF_ADCCTL_CMD_EN;
126  }
127 
128  // Release reset and enable the ADC
129  ADI8BitsSet(AUX_ADI4_BASE, ADI_4_AUX_O_ADC0, ADI_4_AUX_ADC0_EN_M | ADI_4_AUX_ADC0_RESET_N_M | sampleTime);
130 }

151
