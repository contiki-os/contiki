312 VIMSLineBufDisable(uint32_t ui32Base)
313 {
314  //
315  // Disable line buffers
316  //
317  HWREG(ui32Base + VIMS_O_CTL) |= VIMS_CTL_IDCODE_LB_DIS_M |
318  VIMS_CTL_SYSBUS_LB_DIS_M;
