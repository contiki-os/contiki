59 void
60 TRNGConfigure(uint32_t ui32MinSamplesPerCycle,
61  uint32_t ui32MaxSamplesPerCycle,
62  uint32_t ui32ClocksPerSample)
63 {
64  uint32_t ui32Val;
65 
66  //
67  // Make sure the TRNG is disabled.
68  //
69  ui32Val = HWREG(TRNG_BASE + TRNG_O_CTL) & ~TRNG_CTL_TRNG_EN;
70  HWREG(TRNG_BASE + TRNG_O_CTL) = ui32Val;
71 
72  //
73  // Configure the startup number of samples.
74  //
75  ui32Val &= ~TRNG_CTL_STARTUP_CYCLES_M;
76  ui32Val |= ((ui32MaxSamplesPerCycle >> 8) & 0xFFFF) << 16;
77  HWREG(TRNG_BASE + TRNG_O_CTL) = ui32Val;
78 
79  //
80  // Configure the minimum and maximum number of samples pr generated number
81  // and the number of clocks per sample.
82  //
83  HWREG(TRNG_BASE + TRNG_O_CFG0) =
84  (((ui32MaxSamplesPerCycle >> 8) & 0xFFFF) << 16) |
85  ((ui32ClocksPerSample & 0xFF) << 8) |
86  ((ui32MinSamplesPerCycle >> 6) & 0xFF);
87 }

94 uint32_t
95 TRNGNumberGet(uint32_t ui32Word)
96 {
97  uint32_t ui32RandomNumber;
98 
99  //
100  // Check the arguments.
101  //
102  ASSERT((ui32Word == TRNG_HI_WORD) ||
103  (ui32Word == TRNG_LOW_WORD));
104 
105  //
106  // Return the right requested part of the generated number.
107  //
108  if(ui32Word == TRNG_HI_WORD)
109  {
110  ui32RandomNumber = HWREG(TRNG_BASE + TRNG_O_OUT1);
111  }
112  else
113  {
114  ui32RandomNumber = HWREG(TRNG_BASE + TRNG_O_OUT0);
115  }
116 
117  //
118  // Initiate generation of new number.
119  //
120  HWREG(TRNG_BASE + TRNG_O_IRQFLAGCLR) = 0x1;
121 
122  //
123  // Return the random number.
124  //
125  return ui32RandomNumber;
126 }
